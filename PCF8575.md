```bash
#include <Wire.h>

#define PCF8575_ADDR 0x20  // 사용자 설정 주소 (주의: 7bit 주소 기준)

void setup() {
  Wire.begin();
  Serial.begin(9600);
  delay(1000);
  Serial.println("모든 핀을 5초마다 HIGH <-> LOW 토글합니다.");
}

void loop() {
  static bool allHigh = false;  // 초기 상태: LOW

  uint16_t out = allHigh ? 0xFFFF : 0x0000;  // HIGH or LOW 설정

  Wire.beginTransmission(PCF8575_ADDR);
  Wire.write(out & 0xFF);         // 하위 바이트 (P07~P00)
  Wire.write((out >> 8) & 0xFF);  // 상위 바이트 (P17~P10)
  byte error = Wire.endTransmission();

  if (error == 0) {
    Serial.print("모든 핀 상태: ");
    Serial.println(allHigh ? "HIGH" : "LOW");
  } else {
    Serial.print("I2C 통신 오류: ");
    Serial.println(error);
  }

  allHigh = !allHigh;  // 상태 반전
  delay(5000);         // 10초 대기
}
```

```bash
#include <Wire.h>

const uint8_t PCF8575_ADDR = 0x20;  // 보통 기본 주소 (A2~A0 = GND)

void setup() {
  Wire.begin();  // I2C 시작

  writePCF8575(0x00);  
  delay(1000);

  // P07 = HIGH 로 설정 (0000 0001)
  writePCF8575(0x01);  // LSB가 P07
}

void loop() {
  // 아무것도 안 함
}

// 16비트 데이터 출력 함수
void writePCF8575(uint16_t data) {
  Wire.beginTransmission(PCF8575_ADDR);
  Wire.write(lowByte(data));   // P07~P00
  Wire.write(highByte(data));  // P17~P10
  Wire.endTransmission();
}
```
